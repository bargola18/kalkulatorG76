function hitung(){ let D = parseFloat(document.getElementById('dia').value); let P = parseFloat(document.getElementById('pitch').value); let L = parseFloat(document.getElementById('length').value); let unit = document.getElementById('unit').value; let type = document.getElementById('type').value; let rumus = parseFloat(document.getElementById('rumus').value); let mesin = document.getElementById('mesin').value; if(!D ||!P ||!L){alert('Lengkapi Diameter, Pitch, Panjang');return} let H = rumus * P; let X = type=='od'? D - 2*H : D + 2*H; let R = 0.; let gcode = ''; let Q1, Q2, P_mic; // 1. GSK 980-TD = Default: Baris1 Q1000 | Baris2 Q200 FIX | P x10000 if(mesin=='gsk980td'){ Q1 = 1000; Q2 = 200; // <-- FIX 200, bukan dihitung P_mic = Math.round(H*10000); // P 4 digit: 1.240 -> 12400 gcode = `G76 P030060 Q${Q1} R${R}\n`; } // 2. GSK 980-TDHi: Q baris 1 & 2 fix 0.2 else if(mesin=='gsk980tdhi'){ Q1 = Q2 = 0.2; P_mic = Math.round(H*1000); gcode = `G76 P030060 Q${Q1} R${R}\n`; } // 3. Fanuc Oi-Mate-TD: Q baris 1 & 2 fix 200 else if(mesin=='fanuc0i'){ Q1 = Q2 = 200; P_mic = Math.round(H*1000); gcode = `G76 P030060 Q${Q1} R${R}\n`; } gcode += `G76 X${X.toFixed(3)} Z-${L} P${P_mic} Q${Q2} F${P}`; if(unit=='inch'){ gcode = 'G20 ; Inch\n' + gcode; } else { gcode = 'G21 ; Metric\n' + gcode; } document.getElementById('output').textContent = gcode; }
